# GitHub Upload Commands

# 1. Open Command Prompt or PowerShell
# 2. Navigate to your project folder
cd c:\Users\aaary\Surf\trust-website-github

# 3. Initialize Git repository
git init

# 4. Add all files
git add .

# 5. Make first commit
git commit -m "Initial commit - Trust website"

# 6. Add GitHub remote (replace YOUR_USERNAME with your GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/choudhary-rafal-singh-trust.git

# 7. Push to GitHub
git branch -M main
git push -u origin main
